<interfaces>
useResponsiveStyles() returns: { padding, fontSize, spacing, borderRadius, layout }
- padding: { horizontal, vertical, card }
- fontSize: { title, cardTitle, body, small }
- spacing: { cardGap, component }
- borderRadius: { card, large }
- layout: { maxContentWidth, gridColumns, gridItemWidth, gridGap }

createResponsiveMetrics(width, height) returns ResponsiveMetrics with same shape.

responsiveUtils mapping: responsivePadding.horizontal(base) → padding.horizontal; responsiveFontSize.title(base) → fontSize.title; responsiveSpacing.cardGap() → spacing.cardGap; responsiveBorderRadius.card() → borderRadius.card; responsiveGrid.columns() → layout.gridColumns; responsiveGrid.itemWidth(base) → layout.gridItemWidth; responsiveGrid.gap() → layout.gridGap
</interfaces>
